\begin{exercise}[Nullmengen]
	Für ein beliebiges äußeres Maß \(\mu^*\) gilt, alle Nullmengen,
	\[
		A\subseteq \Omega : \mu^*(A) = 0,
	\]
	sind \(\mu^*\)-messbar.
\end{exercise}
